        \item \textbf{Max Queue Length}\\

% ------ Frame 3: Phân tích Hàng đợi ------
\begin{frame}{Phân tích: Hàng đợi và Thời gian chờ}
  \begin{columns}[T]
    \column{0.50\textwidth}
    \begin{block}{Hàng đợi trung bình (đã nhân trọng số)}
      \begin{itemize}
        \item DQN: \textbf{16.05 xe}
        \item Fixed-Time: \textbf{33.13 xe}
        \item[$\Rightarrow$] \alert{Giảm 51.6\%}
      \end{itemize}
      Độ lệch chuẩn: 8.16 vs 12.77\\
      $\Rightarrow$ DQN \textbf{ổn định hơn 36.1\%}\\
      (ít đợt ùn tắc cục bộ đột ngột)
    \end{block}
    \vspace{6pt}
    \begin{block}{Queue-delay (proxy)}
      \begin{itemize}
        \item DQN: \textbf{30.55}
        \item Fixed-Time: \textbf{121.35}
        \item[$\Rightarrow$] \alert{Giảm 74.8\%}
      \end{itemize}
      Cải thiện lớn nhất trong tất cả chỉ số
    \end{block}

    \column{0.46\textwidth}
    \begin{exampleblock}{Lý do DQN tốt hơn}
      \begin{itemize}
        \item Quan sát $q_i$, $w_i$ liên tục theo thời gian thực
        \item \textbf{Tập trung thời gian xanh} vào hướng đang tích lũy xe
        \item Không lãng phí thời gian xanh khi hướng đó vắng xe
      \end{itemize}
    \end{exampleblock}
    \vspace{6pt}
    \begin{alertblock}{Fixed-Time không thể làm điều này}
      Dù NS tắc hay EW trống, đèn vẫn giữ đúng \textbf{100s / 50s cố định} --- không phản ứng với thực tế
    \end{alertblock}
  \end{columns}
\end{frame}

% ------ Frame 4: Phân tích Tốc độ & Throughput ------
\begin{frame}{Phân tích: Tốc độ, Throughput và Reward}
  \begin{columns}[T]
    \column{0.50\textwidth}
    \begin{block}{Tốc độ lưu thông}
      DQN: \textbf{5.65 m/s} \quad Fixed-Time: \textbf{4.87 m/s}\\
      $\Rightarrow$ \textcolor{techgreen}{\textbf{Tăng 16.0\%}}\\[4pt]
      Xe di chuyển nhanh hơn → dòng chảy thông thoáng hơn tổng thể
    \end{block}
    \vspace{6pt}
    \begin{block}{Throughput}
      DQN: \textbf{344 xe} \quad Fixed-Time: \textbf{338 xe}\\
      $\Rightarrow$ \textcolor{techgreen}{\textbf{Tăng 1.8\%}} \quad (khiêm tốn)\\[4pt]
      Cả hai xử lý lượng xe tương đương, nhưng DQN làm với \textbf{ít ùn tắc và chờ đợi} hơn đáng kể
    \end{block}
    \vspace{6pt}
    \begin{block}{Tổng Reward}
      DQN: $-$14.754 \quad Fixed-Time: $-$35.512\\
      $\Rightarrow$ \textcolor{techgreen}{\textbf{Cải thiện 58.5\%}}\\
      Agent đã tối ưu hóa thành công hàm mục tiêu
    \end{block}

    \column{0.46\textwidth}
    \begin{exampleblock}{Chiến lược DQN học được}
      \begin{itemize}
        \setlength{\itemsep}{5pt}
        \item \textbf{Ưu tiên làn hàng đợi dài}:\\
              Tập trung xanh cho hướng đang tắc
        \item \textbf{Giữ EW $\geq$ 60s}:\\
              Đủ thời gian xử lý trục chính (lưu lượng gấp đôi)
        \item \textbf{Không bỏ quên NS}:\\
              $t_{\max}=70$s ép chuyển pha đúng lúc
        \item \textbf{Tránh chuyển sớm}:\\
              Học tránh penalty $-1.0$ → kiên nhẫn giữ pha
      \end{itemize}
    \end{exampleblock}
  \end{columns}
\end{frame}
